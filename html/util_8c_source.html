46 /* imap_expand_path: IMAP implementation of mutt_expand_path. Rewrite
47  * an IMAP path in canonical and absolute form.
48  * Inputs: a buffer containing an IMAP path, and the number of bytes in
49  * that buffer.
50  * Outputs: The buffer is rewritten in place with the canonical IMAP path.
51  * Returns 0 on success, or -1 if imap_parse_path chokes or url_ciss_tostring
52  * fails, which it might if there isn't enough room in the buffer. */
53 int imap_expand_path (char* path, size_t len)
54 {
55  IMAP_MBOX mx;
56  IMAP_DATA* idata;
57  ciss_url_t url;
58  char fixedpath[LONG_STRING];
59  int rc;
60 
61  if (imap_parse_path (path, &mx) < 0)
62  return -1;
63 
64  idata = imap_conn_find (&mx.account, MUTT_IMAP_CONN_NONEW);
65  mutt_account_tourl (&mx.account, &url);
66  imap_fix_path (idata, mx.mbox, fixedpath, sizeof (fixedpath));
67  url.path = fixedpath;
68 
69  rc = url_ciss_tostring (&url, path, len, U_DECODE_PASSWD);
70  FREE (&mx.mbox);
71 
72  return rc;
73 }

75 /* Public function
76  *
77  * Provided an imap mbox name and a delimiter, returns the mbox parent
78  * name.
79  *
80  * Could be static with a prototype in imap_private.h, but could be useful
81  * as a public function.
82  */
83 void imap_get_parent (char *output, const char *mbox, size_t olen, char delim)
84 {
85  int n;
86 
87  /* Make a copy of the mailbox name, but only if the pointers are different */
88  if (mbox != output)
89  strfcpy (output, mbox, olen);
90 
91  n = mutt_strlen (output);
92 
93  /* Let's go backwards until the next delimiter
94  *
95  * If output[n] is a '/', the first n-- will allow us
96  * to ignore it. If it isn't, then output looks like
97  * "/aaaaa/bbbb". There is at least one "b", so we can't skip
98  * the "/" after the 'a's.
99  *
100  * If output == '/', then n-- => n == 0, so the loop ends
101  * immediately
102  */
103  for (n--; n >= 0 && output[n] != delim ; n--);
104 
105  /* We stopped before the beginning. There is a trailing
106  * slash.
107  */
108  if (n > 0)
109  {
110  /* Strip the trailing delimiter. */
111  output[n] = '\0';
112  }
113  else
114  {
115  output[0] = (n == 0) ? delim : '\0';
116  }
117 }

265 /* imap_parse_path: given an IMAP mailbox name, return host, port
266  * and a path IMAP servers will recognize.
267  * mx.mbox is malloc'd, caller must free it */
268 int imap_parse_path (const char* path, IMAP_MBOX* mx)
269 {
270  static unsigned short ImapPort = 0;
271  static unsigned short ImapsPort = 0;
272  struct servent* service;
273  char tmp[128];
274  ciss_url_t url;
275  char *c;
276  int n;
277 
278  if (!ImapPort)
279  {
280  service = getservbyname ("imap", "tcp");
281  if (service)
282  ImapPort = ntohs (service->s_port);
283  else
284  ImapPort = IMAP_PORT;
285  dprint (3, (debugfile, "Using default IMAP port %d\n", ImapPort));
286  }
287  if (!ImapsPort)
288  {
289  service = getservbyname ("imaps", "tcp");
290  if (service)
291  ImapsPort = ntohs (service->s_port);
292  else
293  ImapsPort = IMAP_SSL_PORT;
294  dprint (3, (debugfile, "Using default IMAPS port %d\n", ImapsPort));
295  }
296 
297  /* Defaults */
298  memset(&mx->account, 0, sizeof(mx->account));
299  mx->account.port = ImapPort;
300  mx->account.type = MUTT_ACCT_TYPE_IMAP;
301 
302  c = safe_strdup (path);
303  url_parse_ciss (&url, c);
304  if (url.scheme == U_IMAP || url.scheme == U_IMAPS)
305  {
306  if (mutt_account_fromurl (&mx->account, &url) < 0 || !*mx->account.host)
307  {
308  FREE (&c);
309  return -1;
310  }
311 
312  mx->mbox = safe_strdup (url.path);
313 
314  if (url.scheme == U_IMAPS)
315  mx->account.flags |= MUTT_ACCT_SSL;
316 
317  FREE (&c);
318  }
319  /* old PINE-compatibility code */
320  else
321  {
322  FREE (&c);
323  if (sscanf (path, "{%127[^}]}", tmp) != 1)
324  return -1;
325 
326  c = strchr (path, '}');
327  if (!c)
328  return -1;
329  else
330  /* walk past closing '}' */
331  mx->mbox = safe_strdup (c+1);
332 
333  if ((c = strrchr (tmp, '@')))
334  {
335  *c = '\0';
336  strfcpy (mx->account.user, tmp, sizeof (mx->account.user));
337  strfcpy (tmp, c+1, sizeof (tmp));
338  mx->account.flags |= MUTT_ACCT_USER;
339  }
340 
341  if ((n = sscanf (tmp, "%127[^:/]%127s", mx->account.host, tmp)) < 1)
342  {
343  dprint (1, (debugfile, "imap_parse_path: NULL host in %s\n", path));
344  FREE (&mx->mbox);
345  return -1;
346  }
347 
348  if (n > 1) {
349  if (sscanf (tmp, ":%hu%127s", &(mx->account.port), tmp) >= 1)
350  mx->account.flags |= MUTT_ACCT_PORT;
351  if (sscanf (tmp, "/%s", tmp) == 1)
352  {
353  if (!ascii_strncmp (tmp, "ssl", 3))
354  mx->account.flags |= MUTT_ACCT_SSL;
355  else
356  {
357  dprint (1, (debugfile, "imap_parse_path: Unknown connection type in %s\n", path));
358  FREE (&mx->mbox);
359  return -1;
360  }
361  }
362  }
363  }
364 
365  if ((mx->account.flags & MUTT_ACCT_SSL) && !(mx->account.flags & MUTT_ACCT_PORT))
366  mx->account.port = ImapsPort;
367 
368  return 0;
369 }

398 /* imap_pretty_mailbox: called by mutt_pretty_mailbox to make IMAP paths
399  * look nice. */
400 void imap_pretty_mailbox (char* path)
401 {
402  IMAP_MBOX home, target;
403  ciss_url_t url;
404  char* delim;
405  int tlen;
406  int hlen = 0;
407  char home_match = 0;
408 
409  if (imap_parse_path (path, &target) < 0)
410  return;
411 
412  tlen = mutt_strlen (target.mbox);
413  /* check whether we can do '=' substitution */
414  if (mx_is_imap(Maildir) && !imap_parse_path (Maildir, &home))
415  {
416  hlen = mutt_strlen (home.mbox);
417  if (tlen && mutt_account_match (&home.account, &target.account) &&
418  !mutt_strncmp (home.mbox, target.mbox, hlen))
419  {
420  if (! hlen)
421  home_match = 1;
422  else if (ImapDelimChars)
423  for (delim = ImapDelimChars; *delim != '\0'; delim++)
424  if (target.mbox[hlen] == *delim)
425  home_match = 1;
426  }
427  FREE (&home.mbox);
428  }
429 
430  /* do the '=' substitution */
431  if (home_match) {
432  *path++ = '=';
433  /* copy remaining path, skipping delimiter */
434  if (! hlen)
435  hlen = -1;
436  memcpy (path, target.mbox + hlen + 1, tlen - hlen - 1);
437  path[tlen - hlen - 1] = '\0';
438  }
439  else
440  {
441  mutt_account_tourl (&target.account, &url);
442  url.path = target.mbox;
443  /* FIXME: That hard-coded constant is bogus. But we need the actual
444  * size of the buffer from mutt_pretty_mailbox. And these pretty
445  * operations usually shrink the result. Still... */
446  url_ciss_tostring (&url, path, 1024, 0);
447  }
448 
449  FREE (&target.mbox);
450 }

507 /*
508  * Fix up the imap path. This is necessary because the rest of mutt
509  * assumes a hierarchy delimiter of '/', which is not necessarily true
510  * in IMAP. Additionally, the filesystem converts multiple hierarchy
511  * delimiters into a single one, ie "///" is equal to "/". IMAP servers
512  * are not required to do this.
513  * Moreover, IMAP servers may dislike the path ending with the delimiter.
514  */
515 char *imap_fix_path (IMAP_DATA *idata, const char *mailbox, char *path,
516  size_t plen)
517 {
518  int i = 0;
519  char delim = '\0';
520 
521  if (idata)
522  delim = idata->delim;
523 
524  while (mailbox && *mailbox && i < plen - 1)
525  {
526  if ((ImapDelimChars && strchr(ImapDelimChars, *mailbox))
527  || (delim && *mailbox == delim))
528  {
529  /* use connection delimiter if known. Otherwise use user delimiter */
530  if (!idata)
531  delim = *mailbox;
532 
533  while (*mailbox
534  && ((ImapDelimChars && strchr(ImapDelimChars, *mailbox))
535  || (delim && *mailbox == delim)))
536  mailbox++;
537  path[i] = delim;
538  }
539  else
540  {
541  path[i] = *mailbox;
542  mailbox++;
543  }
544  i++;
545  }
546  if (i && path[--i] != delim)
547  i++;
548  path[i] = '\0';
549 
550  return path;
551 }

554  size_t dlen)

570  }
571  else

573  p++;
574  s++;
575  }
576  *s = '\0';
577 }

613 /* imap_next_word: return index into string where next IMAP word begins */
614 char *imap_next_word (char *s)
615 {
616  int quoted = 0;
617 
618  while (*s) {
619  if (*s == '\\') {
620  s++;
621  if (*s)
622  s++;
623  continue;
624  }
625  if (*s == '\"')
626  quoted = quoted ? 0 : 1;
627  if (!quoted && ISSPACE (*s))

629  s++;
630  }

636 /* imap_parse_date: date is of the form: DD-MMM-YYYY HH:MM:SS +ZZzz */
637 time_t imap_parse_date (char *s)
638 {
639  struct tm t;
640  time_t tz;
641 
642  t.tm_mday = (s[0] == ' '? s[1] - '0' : (s[0] - '0') * 10 + (s[1] - '0'));
643  s += 2;
644  if (*s != '-')
645  return 0;
646  s++;
647  t.tm_mon = mutt_check_month (s);
648  s += 3;
649  if (*s != '-')
650  return 0;
651  s++;
652  t.tm_year = (s[0] - '0') * 1000 + (s[1] - '0') * 100 + (s[2] - '0') * 10 + (s[3] - '0') - 1900;
653  s += 4;
654  if (*s != ' ')
655  return 0;
656  s++;
657 
658  /* time */
659  t.tm_hour = (s[0] - '0') * 10 + (s[1] - '0');
660  s += 2;
661  if (*s != ':')
662  return 0;
663  s++;
664  t.tm_min = (s[0] - '0') * 10 + (s[1] - '0');
665  s += 2;
666  if (*s != ':')
667  return 0;
668  s++;
669  t.tm_sec = (s[0] - '0') * 10 + (s[1] - '0');
670  s += 2;
671  if (*s != ' ')
672  return 0;
673  s++;
674 
675  /* timezone */
676  tz = ((s[1] - '0') * 10 + (s[2] - '0')) * 3600 +
677  ((s[3] - '0') * 10 + (s[4] - '0')) * 60;
678  if (s[0] == '+')
679  tz = -tz;
680 
681  return (mutt_mktime (&t, 0) + tz);
682 }

713  * surround string with quotes, escape " and \ with \ */
714 void imap_quote_string (char *dest, size_t dlen, const char *src)
715 {
716  static const char quote[] = "\"\\";
717  char *pt;
718  const char *s;
719 
720  pt = dest;
721  s = src;
722 
723  *pt++ = '"';
724  /* save room for trailing quote-char */
725  dlen -= 2;
726 
727  for (; *s && dlen; s++)
728  {
729  if (strchr (quote, *s))
730  {
731  dlen -= 2;
732  if (!dlen)
733  break;
734  *pt++ = '\\';
735  *pt++ = *s;
736  }
737  else
738  {
739  *pt++ = *s;
740  dlen--;
741  }
742  }
743  *pt++ = '"';
744  *pt = 0;
745 }

844 void imap_keepalive (void)
845 {
846  CONNECTION *conn;
847  CONTEXT *ctx = NULL;
848  IMAP_DATA *idata;
849 
850  conn = mutt_socket_head ();
851  while (conn)
852  {
853  if (conn->account.type == MUTT_ACCT_TYPE_IMAP)
854  {
855  int need_free = 0;
856 
857  idata = (IMAP_DATA*) conn->data;
858 
859  if (idata->state >= IMAP_AUTHENTICATED
860  && time(NULL) >= idata->lastread + ImapKeepalive)
861  {
862  if (idata->ctx)
863  ctx = idata->ctx;
864  else
865  {
866  ctx = safe_calloc (1, sizeof (CONTEXT));
867  ctx->data = idata;
868  /* imap_close_mailbox will set ctx->iadata->ctx to NULL, so we can't
869  * rely on the value of iadata->ctx to determine if this placeholder
870  * context needs to be freed.
871  */
872  need_free = 1;
873  }
874  /* if the imap connection closes during this call, ctx may be invalid
875  * after this point, and thus should not be read.
876  */
877  imap_check_mailbox (ctx, NULL, 1);
878  if (need_free)
879  FREE (&ctx);
880  }
881  }
882 
883  conn = conn->next;
884  }
885 }

887 int imap_wait_keepalive (pid_t pid)
888 {
889  struct sigaction oldalrm;
890  struct sigaction act;
891  sigset_t oldmask;
892  int rc;
893 
894  short imap_passive = option (OPTIMAPPASSIVE);
895 
896  set_option (OPTIMAPPASSIVE);
897  set_option (OPTKEEPQUIET);
898 
899  sigprocmask (SIG_SETMASK, NULL, &oldmask);
900 
901  sigemptyset (&act.sa_mask);
902  act.sa_handler = alrm_handler;
903 #ifdef SA_INTERRUPT
904  act.sa_flags = SA_INTERRUPT;
905 #else
906  act.sa_flags = 0;
907 #endif
908 
909  sigaction (SIGALRM, &act, &oldalrm);
910 
911  alarm (ImapKeepalive);
912  while (waitpid (pid, &rc, 0) < 0 && errno == EINTR)
913  {
914  alarm (0); /* cancel a possibly pending alarm */
915  imap_keepalive ();
916  alarm (ImapKeepalive);
917  }
918 
919  alarm (0); /* cancel a possibly pending alarm */
920 
921  sigaction (SIGALRM, &oldalrm, NULL);
922  sigprocmask (SIG_SETMASK, &oldmask, NULL);
923 
924  unset_option (OPTKEEPQUIET);
925  if (!imap_passive)
926  unset_option (OPTIMAPPASSIVE);
927 
928  return rc;
929 }
