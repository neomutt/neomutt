153 static int dotlock_file (const char *path, int fd, int retry)
154 {
155  int r;
156  int flags = DL_FL_USEPRIV | DL_FL_RETRY;
157 
158  if (retry) retry = 1;
159 
160 retry_lock:
161  if ((r = invoke_dotlock(path, fd, flags, retry)) == DL_EX_EXIST)
162  {
163  if (!option (OPTNOCURSES))
164  {
165  char msg[LONG_STRING];
166 
167  snprintf(msg, sizeof(msg), _("Lock count exceeded, remove lock for %s?"),
168  path);
169  if(retry && mutt_yesorno(msg, MUTT_YES) == MUTT_YES)
170  {
171  flags |= DL_FL_FORCE;
172  retry--;
173  mutt_clear_error ();
174  goto retry_lock;
175  }
176  }
177  else
178  {
179  mutt_error ( _("Can't dotlock %s.\n"), path);
180  }
181  }
182  return (r == DL_EX_OK ? 0 : -1);
183 }

193 /* Args:
194  * excl if excl != 0, request an exclusive lock
195  * dot if dot != 0, try to dotlock the file
196  * timeout should retry locking?
197  */
198 int mx_lock_file (const char *path, int fd, int excl, int dot, int timeout)
199 {
200 #if defined (USE_FCNTL) || defined (USE_FLOCK)
201  int count;
202  int attempt;
203  struct stat sb = { 0 }, prev_sb = { 0 }; /* silence gcc warnings */
204 #endif
205  int r = 0;
206 
207 #ifdef USE_FCNTL
208  struct flock lck;
209 
210  memset (&lck, 0, sizeof (struct flock));
211  lck.l_type = excl ? F_WRLCK : F_RDLCK;
212  lck.l_whence = SEEK_SET;
213 
214  count = 0;
215  attempt = 0;
216  while (fcntl (fd, F_SETLK, &lck) == -1)
217  {
218  dprint(1,(debugfile, "mx_lock_file(): fcntl errno %d.\n", errno));
219  if (errno != EAGAIN && errno != EACCES)
220  {
221  mutt_perror ("fcntl");
222  return -1;
223  }
224 
225  if (fstat (fd, &sb) != 0)
226  sb.st_size = 0;
227 
228  if (count == 0)
229  prev_sb = sb;
230 
231  /* only unlock file if it is unchanged */
232  if (prev_sb.st_size == sb.st_size && ++count >= (timeout?MAXLOCKATTEMPT:0))
233  {
234  if (timeout)
235  mutt_error _("Timeout exceeded while attempting fcntl lock!");
236  return -1;
237  }
238 
239  prev_sb = sb;
240 
241  mutt_message (_("Waiting for fcntl lock... %d"), ++attempt);
242  sleep (1);
243  }
244 #endif /* USE_FCNTL */
245 
246 #ifdef USE_FLOCK
247  count = 0;
248  attempt = 0;
249  while (flock (fd, (excl ? LOCK_EX : LOCK_SH) | LOCK_NB) == -1)
250  {
251  if (errno != EWOULDBLOCK)
252  {
253  mutt_perror ("flock");
254  r = -1;
255  break;
256  }
257 
258  if (fstat(fd, &sb) != 0)
259  sb.st_size = 0;
260 
261  if (count == 0)
262  prev_sb = sb;
263 
264  /* only unlock file if it is unchanged */
265  if (prev_sb.st_size == sb.st_size && ++count >= (timeout?MAXLOCKATTEMPT:0))
266  {
267  if (timeout)
268  mutt_error _("Timeout exceeded while attempting flock lock!");
269  r = -1;
270  break;
271  }
272 
273  prev_sb = sb;
274 
275  mutt_message (_("Waiting for flock attempt... %d"), ++attempt);
276  sleep (1);
277  }
278 #endif /* USE_FLOCK */
279 
280 #ifdef USE_DOTLOCK
281  if (r == 0 && dot)
282  r = dotlock_file (path, fd, timeout);
283 #endif /* USE_DOTLOCK */
284 
285  if (r != 0)
286  {
287  /* release any other locks obtained in this routine */
288 
289 #ifdef USE_FCNTL
290  lck.l_type = F_UNLCK;
291  fcntl (fd, F_SETLK, &lck);
292 #endif /* USE_FCNTL */
293 
294 #ifdef USE_FLOCK
295  flock (fd, LOCK_UN);
296 #endif /* USE_FLOCK */
297  }
298 
299  return r;
300 }

431 int mx_get_magic (const char *path)
432 {
433  struct stat st;
434  int magic = 0;
435  char tmp[_POSIX_PATH_MAX];
436  FILE *f;
437 
438 #ifdef USE_IMAP
439  if(mx_is_imap(path))
440  return MUTT_IMAP;
441 #endif /* USE_IMAP */
442 
443 #ifdef USE_POP
444  if (mx_is_pop (path))
445  return MUTT_POP;
446 #endif /* USE_POP */
447 
448 #ifdef USE_NNTP
449  if (mx_is_nntp (path))
450  return MUTT_NNTP;
451 #endif /* USE_NNTP */
452 
453 #ifdef USE_NOTMUCH
454  if (mx_is_notmuch(path))
455  return MUTT_NOTMUCH;
456 #endif
457 
458  if (stat (path, &st) == -1)
459  {
460  dprint (1, (debugfile, "mx_get_magic(): unable to stat %s: %s (errno %d).\n",
461  path, strerror (errno), errno));
462  return (-1);
463  }
464 
465  if (S_ISDIR (st.st_mode))
466  {
467  /* check for maildir-style mailbox */
468  if (mx_is_maildir (path))
469  return MUTT_MAILDIR;
470 
471  /* check for mh-style mailbox */
472  if (mx_is_mh (path))
473  return MUTT_MH;
474  }
475  else if (st.st_size == 0)
476  {
477  /* hard to tell what zero-length files are, so assume the default magic */
478  if (DefaultMagic == MUTT_MBOX || DefaultMagic == MUTT_MMDF)
479  return (DefaultMagic);
480  else
481  return (MUTT_MBOX);
482  }
483  else if ((f = fopen (path, "r")) != NULL)
484  {
485  struct utimbuf times;
486 
487  fgets (tmp, sizeof (tmp), f);
488  if (mutt_strncmp ("From ", tmp, 5) == 0)
489  magic = MUTT_MBOX;
490  else if (mutt_strcmp (MMDF_SEP, tmp) == 0)
491  magic = MUTT_MMDF;
492  safe_fclose (&f);
493 
494  if (!option(OPTCHECKMBOXSIZE))
495  {
496  /* need to restore the times here, the file was not really accessed,
497  * only the type was accessed. This is important, because detection
498  * of "new mail" depends on those times set correctly.
499  */
500  times.actime = st.st_atime;
501  times.modtime = st.st_mtime;
502  utime (path, &times);
503  }
504  }
505  else
506  {
507  dprint (1, (debugfile, "mx_get_magic(): unable to open file %s for reading.\n",
508  path));
509  return (-1);
510  }
511 
512 #ifdef USE_COMPRESSED
513  /* If there are no other matches, see if there are any
514  * compress hooks that match */
515  if ((magic == 0) && mutt_comp_can_read (path))
516  return MUTT_COMPRESSED;
517 #endif
518  return (magic);
519 }

596 /*
597  * open a mailbox and parse it
598  *
599  * Args:
600  * flags MUTT_NOSORT do not sort mailbox
601  * MUTT_APPEND open mailbox for appending
602  * MUTT_READONLY open mailbox in read-only mode
603  * MUTT_QUIET only print error messages
604  * MUTT_PEEK revert atime where applicable
605  * ctx if non-null, context struct to use
606  */
607 CONTEXT *mx_open_mailbox (const char *path, int flags, CONTEXT *pctx)
608 {
609  CONTEXT *ctx = pctx;
610  int rc;
611 
612  if (!path || !path[0])
613  return NULL;
614 
615  if (!ctx)
616  ctx = safe_malloc (sizeof (CONTEXT));
617  memset (ctx, 0, sizeof (CONTEXT));
618  ctx->path = safe_strdup (path);
619  if (! (ctx->realpath = realpath (ctx->path, NULL)) )
620  ctx->realpath = safe_strdup (ctx->path);
621 
622  ctx->msgnotreadyet = -1;
623  ctx->collapsed = 0;
624 
625  for (rc=0; rc < RIGHTSMAX; rc++)
626  mutt_bit_set(ctx->rights,rc);
627 
628  if (flags & MUTT_QUIET)
629  ctx->quiet = 1;
630  if (flags & MUTT_READONLY)
631  ctx->readonly = 1;
632  if (flags & MUTT_PEEK)
633  ctx->peekonly = 1;
634 
635  if (flags & (MUTT_APPEND|MUTT_NEWFOLDER))
636  {
637  if (mx_open_mailbox_append (ctx, flags) != 0)
638  {
639  mx_fastclose_mailbox (ctx);
640  if (!pctx)
641  FREE (&ctx);
642  return NULL;
643  }
644  return ctx;
645  }
646 
647  ctx->magic = mx_get_magic (path);
648  ctx->mx_ops = mx_get_ops (ctx->magic);
649 
650  if (ctx->magic <= 0 || !ctx->mx_ops)
651  {
652  if (ctx->magic == 0 || !ctx->mx_ops)
653  mutt_error (_("%s is not a mailbox."), path);
654  else if (ctx->magic == -1)
655  mutt_perror(path);
656 
657  mx_fastclose_mailbox (ctx);
658  if (!pctx)
659  FREE (&ctx);
660  return (NULL);
661  }
662 
663  /* if the user has a `push' command in their .muttrc, or in a folder-hook,
664  * it will cause the progress messages not to be displayed because
665  * mutt_refresh() will think we are in the middle of a macro. so set a
666  * flag to indicate that we should really refresh the screen.
667  */
668  set_option (OPTFORCEREFRESH);
669 
670  if (!ctx->quiet)
671  mutt_message (_("Reading %s..."), ctx->path);
672 
673  rc = ctx->mx_ops->open(ctx);
674 
675  if (rc == 0)
676  {
677  if ((flags & MUTT_NOSORT) == 0)
678  {
679  /* avoid unnecessary work since the mailbox is completely unthreaded
680  to begin with */
681  unset_option (OPTSORTSUBTHREADS);
682  unset_option (OPTNEEDRESCORE);
683  mutt_sort_headers (ctx, 1);
684  }
685  if (!ctx->quiet)
686  mutt_clear_error ();
687  }
688  else
689  {
690  mx_fastclose_mailbox (ctx);
691  if (!pctx)
692  FREE (&ctx);
693  }
694 
695  unset_option (OPTFORCEREFRESH);
696  return (ctx);
697 }

699 /* free up memory associated with the mailbox context */
700 void mx_fastclose_mailbox (CONTEXT *ctx)
701 {
702  int i;
703  struct utimbuf ut;
704 
705  if(!ctx)
706  return;
707 
708  /* fix up the times so buffy won't get confused */
709  if (ctx->peekonly && ctx->path && (ctx->mtime > ctx->atime)) {
710  ut.actime = ctx->atime;
711  ut.modtime = ctx->mtime;
712  utime (ctx->path, &ut);
713  }
714 
715  /* never announce that a mailbox we've just left has new mail. #3290
716  * XXX: really belongs in mx_close_mailbox, but this is a nice hook point */
717  if (!ctx->peekonly)
718  mutt_buffy_setnotified(ctx->path);
719 
720  if (ctx->mx_ops)
721  ctx->mx_ops->close (ctx);
722 
723 #ifdef USE_COMPRESSED
724  mutt_free_compress_info (ctx);
725 #endif /* USE_COMPRESSED */
726 
727  if (ctx->subj_hash)
728  hash_destroy (&ctx->subj_hash, NULL);
729  if (ctx->id_hash)
730  hash_destroy (&ctx->id_hash, NULL);
731  mutt_clear_threads (ctx);
732  for (i = 0; i < ctx->msgcount; i++)
733  mutt_free_header (&ctx->hdrs[i]);
734  FREE (&ctx->hdrs);
735  FREE (&ctx->v2r);
736  FREE (&ctx->path);
737  FREE (&ctx->realpath);
738  FREE (&ctx->pattern);
739  if (ctx->limit_pattern)
740  mutt_pattern_free (&ctx->limit_pattern);
741  safe_fclose (&ctx->fp);
742  memset (ctx, 0, sizeof (CONTEXT));
743 }
