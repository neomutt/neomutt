146 int mutt_extract_token (BUFFER *dest, BUFFER *tok, int flags)
147 {
148  char ch;
149  char qc = 0; /* quote char */
150  char *pc;
151 
152  /* reset the destination pointer to the beginning of the buffer */
153  dest->dptr = dest->data;
154 
155  SKIPWS (tok->dptr);
156  while ((ch = *tok->dptr))
157  {
158  if (!qc)
159  {
160  if ((ISSPACE (ch) && !(flags & MUTT_TOKEN_SPACE)) ||
161  (ch == '#' && !(flags & MUTT_TOKEN_COMMENT)) ||
162  (ch == '=' && (flags & MUTT_TOKEN_EQUAL)) ||
163  (ch == ';' && !(flags & MUTT_TOKEN_SEMICOLON)) ||
164  ((flags & MUTT_TOKEN_PATTERN) && strchr ("~%=!|", ch)))
165  break;
166  }
167 
168  tok->dptr++;
169 
170  if (ch == qc)
171  qc = 0; /* end of quote */
172  else if (!qc && (ch == '\'' || ch == '"') && !(flags & MUTT_TOKEN_QUOTE))
173  qc = ch;
174  else if (ch == '\\' && qc != '\'')
175  {
176  if (!*tok->dptr)
177  return -1; /* premature end of token */
178  switch (ch = *tok->dptr++)
179  {
180  case 'c':
181  case 'C':
182  if (!*tok->dptr)
183  return -1; /* premature end of token */
184  mutt_buffer_addch (dest, (toupper ((unsigned char) *tok->dptr)
185  - '@') & 0x7f);
186  tok->dptr++;
187  break;
188  case 'r':
189  mutt_buffer_addch (dest, '\r');
190  break;
191  case 'n':
192  mutt_buffer_addch (dest, '\n');
193  break;
194  case 't':
195  mutt_buffer_addch (dest, '\t');
196  break;
197  case 'f':
198  mutt_buffer_addch (dest, '\f');
199  break;
200  case 'e':
201  mutt_buffer_addch (dest, '\033');
202  break;
203  default:
204  if (isdigit ((unsigned char) ch) &&
205  isdigit ((unsigned char) *tok->dptr) &&
206  isdigit ((unsigned char) *(tok->dptr + 1)))
207  {
208 
209  mutt_buffer_addch (dest, (ch << 6) + (*tok->dptr << 3) + *(tok->dptr + 1) - 3504);
210  tok->dptr += 2;
211  }
212  else
213  mutt_buffer_addch (dest, ch);
214  }
215  }
216  else if (ch == '^' && (flags & MUTT_TOKEN_CONDENSE))
217  {
218  if (!*tok->dptr)
219  return -1; /* premature end of token */
220  ch = *tok->dptr++;
221  if (ch == '^')
222  mutt_buffer_addch (dest, ch);
223  else if (ch == '[')
224  mutt_buffer_addch (dest, '\033');
225  else if (isalpha ((unsigned char) ch))
226  mutt_buffer_addch (dest, toupper ((unsigned char) ch) - '@');
227  else
228  {
229  mutt_buffer_addch (dest, '^');
230  mutt_buffer_addch (dest, ch);
231  }
232  }
233  else if (ch == '`' && (!qc || qc == '"'))
234  {
235  FILE *fp;
236  pid_t pid;
237  char *cmd, *ptr;
238  size_t expnlen;
239  BUFFER expn;
240  int line = 0;
241 
242  pc = tok->dptr;
243  do {
244  if ((pc = strpbrk (pc, "\\`")))
245  {
246  /* skip any quoted chars */
247  if (*pc == '\\')
248  pc += 2;
249  }
250  } while (pc && *pc != '`');
251  if (!pc)
252  {
253  dprint (1, (debugfile, "mutt_get_token: mismatched backticks\n"));
254  return (-1);
255  }
256  cmd = mutt_substrdup (tok->dptr, pc);
257  if ((pid = mutt_create_filter (cmd, NULL, &fp, NULL)) < 0)
258  {
259  dprint (1, (debugfile, "mutt_get_token: unable to fork command: %s", cmd));
260  FREE (&cmd);
261  return (-1);
262  }
263  FREE (&cmd);
264 
265  tok->dptr = pc + 1;
266 
267  /* read line */
268  mutt_buffer_init (&expn);
269  expn.data = mutt_read_line (NULL, &expn.dsize, fp, &line, 0);
270  safe_fclose (&fp);
271  mutt_wait_filter (pid);
272 
273  /* if we got output, make a new string consisting of the shell output
274  plus whatever else was left on the original line */
275  /* BUT: If this is inside a quoted string, directly add output to
276  * the token */
277  if (expn.data && qc)
278  {
279  mutt_buffer_addstr (dest, expn.data);
280  FREE (&expn.data);
281  }
282  else if (expn.data)
283  {
284  expnlen = mutt_strlen (expn.data);
285  tok->dsize = expnlen + mutt_strlen (tok->dptr) + 1;
286  ptr = safe_malloc (tok->dsize);
287  memcpy (ptr, expn.data, expnlen);
288  strcpy (ptr + expnlen, tok->dptr); /* __STRCPY_CHECKED__ */
289  if (tok->destroy)
290  FREE (&tok->data);
291  tok->data = ptr;
292  tok->dptr = ptr;
293  tok->destroy = 1; /* mark that the caller should destroy this data */
294  ptr = NULL;
295  FREE (&expn.data);
296  }
297  }
298  else if (ch == '$' && (!qc || qc == '"') && (*tok->dptr == '{' || isalpha ((unsigned char) *tok->dptr)))
299  {
300  const char *env = NULL;
301  char *var = NULL;
302  int idx;
303 
304  if (*tok->dptr == '{')
305  {
306  tok->dptr++;
307  if ((pc = strchr (tok->dptr, '}')))
308  {
309  var = mutt_substrdup (tok->dptr, pc);
310  tok->dptr = pc + 1;
311  }
312  }
313  else
314  {
315  for (pc = tok->dptr; isalnum ((unsigned char) *pc) || *pc == '_'; pc++)
316  ;
317  var = mutt_substrdup (tok->dptr, pc);
318  tok->dptr = pc;
319  }
320  if (var)
321  {
322  if ((env = getenv (var)) || (env = myvar_get (var)))
323  mutt_buffer_addstr (dest, env);
324  else if ((idx = mutt_option_index (var)) != -1)
325  {
326  /* expand settable mutt variables */
327  char val[LONG_STRING];
328 
329  if (var_to_string (idx, val, sizeof (val)))
330  mutt_buffer_addstr (dest, val);
331  }
332  FREE (&var);
333  }
334  }
335  else
336  mutt_buffer_addch (dest, ch);
337  }
338  mutt_buffer_addch (dest, 0); /* terminate the string */
339  SKIPWS (tok->dptr);
340  return 0;
341 }

390  if (!str || *str == '\0')

467 static int add_to_spam_list (SPAM_LIST **list, const char *pat, const char *templ, BUFFER *err)
468 {
469  SPAM_LIST *t = NULL, *last = NULL;
470  REGEXP *rx;
471  int n;
472  const char *p;
473 
474  if (!pat || !*pat || !templ)
475  return 0;
476 
477  if (!(rx = mutt_compile_regexp (pat, REG_ICASE)))
478  {
479  snprintf (err->data, err->dsize, _("Bad regexp: %s"), pat);
480  return -1;
481  }
482 
483  /* check to make sure the item is not already on this list */
484  for (last = *list; last; last = last->next)
485  {
486  if (ascii_strcasecmp (rx->pattern, last->rx->pattern) == 0)
487  {
488  /* Already on the list. Formerly we just skipped this case, but
489  * now we're supporting removals, which means we're supporting
490  * re-adds conceptually. So we probably want this to imply a
491  * removal, then do an add. We can achieve the removal by freeing
492  * the template, and leaving t pointed at the current item.
493  */
494  t = last;
495  FREE(&t->template);
496  break;
497  }
498  if (!last->next)
499  break;
500  }
501 
502  /* If t is set, it's pointing into an extant SPAM_LIST* that we want to
503  * update. Otherwise we want to make a new one to link at the list's end.
504  */
505  if (!t)
506  {
507  t = mutt_new_spam_list();
508  t->rx = rx;
509  if (last)
510  last->next = t;
511  else
512  *list = t;
513  }
514 
515  /* Now t is the SPAM_LIST* that we want to modify. It is prepared. */
516  t->template = safe_strdup(templ);
517 
518  /* Find highest match number in template string */
519  t->nmatch = 0;
520  for (p = templ; *p;)
521  {
522  if (*p == '%')
523  {
524  n = atoi(++p);
525  if (n > t->nmatch)
526  t->nmatch = n;
527  while (*p && isdigit((int)*p))
528  ++p;
529  }
530  else
531  ++p;
532  }
533 
534  if (t->nmatch > t->rx->rx->re_nsub)
535  {
536  snprintf (err->data, err->dsize, _("Not enough subexpressions for spam "
537  "template"));
538  remove_from_spam_list(list, pat);
539  return -1;
540  }
541 
542  t->nmatch++; /* match 0 is always the whole expr */
543 
544  return 0;
545 }
