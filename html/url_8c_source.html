54 int url_pct_decode (char *s)
55 {
56  char *d;
57 
58  if (!s)
59  return -1;
60 
61  for (d = s; *s; s++)
62  {
63  if (*s == '%')
64  {
65  if (s[1] && s[2] &&
66  isxdigit ((unsigned char) s[1]) &&
67  isxdigit ((unsigned char) s[2]) &&
68  hexval (s[1]) >= 0 && hexval (s[2]) >= 0)
69  {
70  *d++ = (hexval (s[1]) << 4) | (hexval (s[2]));
71  s += 2;
72  }
73  else
74  return -1;
75  } else
76  *d++ = *s;
77  }
78  *d ='\0';
79  return 0;
80 }

82 url_scheme_t url_check_scheme (const char *s)
83 {
84  char sbuf[STRING];
85  char *t;
86  int i;
87 
88  if (!s || !(t = strchr (s, ':')))
89  return U_UNKNOWN;
90  if ((size_t)(t - s) >= sizeof (sbuf) - 1)
91  return U_UNKNOWN;
92 
93  strfcpy (sbuf, s, t - s + 1);
94  for (t = sbuf; *t; t++)
95  *t = ascii_tolower (*t);
96 
97  if ((i = mutt_getvaluebyname (sbuf, UrlMap)) == -1)
98  return U_UNKNOWN;
99  else
100  return (url_scheme_t) i;
101 }

104 {

115 /* ciss_parse_userhost: fill in components of ciss with info from src. Note
116  * these are pointers into src, which is altered with '\0's. Port of 0
117  * means no port given. */
118 static int ciss_parse_userhost (ciss_url_t *ciss, char *src)
119 {
120  char *t, *p;
121 
122  ciss->user = NULL;
123  ciss->pass = NULL;
124  ciss->host = NULL;
125  ciss->port = 0;
126 
127  if (strncmp (src, "//", 2) != 0)
128  {
129  ciss->path = src;
130  return url_pct_decode (ciss->path);
131  }
132 
133  src += 2;
134 
135  if ((ciss->path = strchr (src, '/')))
136  *ciss->path++ = '\0';
137 
138  if ((t = strrchr (src, '@')))
139  {
140  *t = '\0';
141  if ((p = strchr (src, ':')))
142  {
143  *p = '\0';
144  ciss->pass = p + 1;
145  if (url_pct_decode (ciss->pass) < 0)
146  return -1;
147  }
148  ciss->user = src;
149  if (url_pct_decode (ciss->user) < 0)
150  return -1;
151  src = t + 1;
152  }
153 
154  /* IPv6 literal address. It may contain colons, so set t to start
155  * the port scan after it.
156  */
157  if ((*src == '[') && (t = strchr (src, ']')))
158  {
159  src++;
160  *t++ = '\0';
161  }
162  else
163  t = src;
164 
165  if ((p = strchr (t, ':')))
166  {
167  int t;
168  *p++ = '\0';
169  if (mutt_atoi (p, &t) < 0 || t < 0 || t > 0xffff)
170  return -1;
171  ciss->port = (unsigned short)t;
172  }
173  else
174  ciss->port = 0;
175 
176  ciss->host = src;
177  return url_pct_decode (ciss->host) >= 0 &&
178  (!ciss->path || url_pct_decode (ciss->path) >= 0) ? 0 : -1;
179 }

195 static void url_pct_encode (char *dst, size_t l, const char *src)
196 {
197  static const char *alph = "0123456789ABCDEF";
198 
199  *dst = 0;
200  l--;
201  while (src && *src && l)
202  {
203  if (strchr ("/:%", *src) && l > 3)
204  {
205  *dst++ = '%';
206  *dst++ = alph[(*src >> 4) & 0xf];
207  *dst++ = alph[*src & 0xf];
208  src++;
209  continue;
210  }
211  *dst++ = *src++;
212  }
213  *dst = 0;
214 }

216 /* url_ciss_tostring: output the URL string for a given CISS object. */
217 int url_ciss_tostring (ciss_url_t* ciss, char* dest, size_t len, int flags)
218 {
219  long l;
220 
221  if (ciss->scheme == U_UNKNOWN)
222  return -1;
223 
224  snprintf (dest, len, "%s:", mutt_getnamebyvalue (ciss->scheme, UrlMap));
225 
226  if (ciss->host)
227  {
228  if (!(flags & U_PATH))
229  safe_strcat (dest, len, "//");
230  len -= (l = strlen (dest)); dest += l;
231 
232  if (ciss->user && (ciss->user[0] || !(flags & U_PATH)))
233  {
234  char u[STRING];
235  url_pct_encode (u, sizeof (u), ciss->user);
236 
237  if (flags & U_DECODE_PASSWD && ciss->pass)
238  {
239  char p[STRING];
240  url_pct_encode (p, sizeof (p), ciss->pass);
241  snprintf (dest, len, "%s:%s@", u, p);
242  }
243  else
244  snprintf (dest, len, "%s@", u);
245 
246  len -= (l = strlen (dest)); dest += l;
247  }
248 
249  if (strchr (ciss->host, ':'))
250  snprintf (dest, len, "[%s]", ciss->host);
251  else
252  snprintf (dest, len, "%s", ciss->host);
253 
254  len -= (l = strlen (dest)); dest += l;
255 
256  if (ciss->port)
257  snprintf (dest, len, ":%hu/", ciss->port);
258  else
259  snprintf (dest, len, "/");
260  }
261 
262  if (ciss->path)
263  safe_strcat (dest, len, ciss->path);
264 
265  return 0;
266 }

268 int url_parse_mailto (ENVELOPE *e, char **body, const char *src)
269 {
270  char *t, *p;
271  char *tmp;
272  char *headers;
273  char *tag, *value;
274 
275  int rc = -1;
276 
277  LIST *last = NULL;
278 
279  if (!(t = strchr (src, ':')))
280  return -1;
281 
282  /* copy string for safe use of strtok() */
283  if ((tmp = safe_strdup (t + 1)) == NULL)
284  return -1;
285 
286  if ((headers = strchr (tmp, '?')))
287  *headers++ = '\0';
288 
289  if (url_pct_decode (tmp) < 0)
290  goto out;
291 
292  e->to = rfc822_parse_adrlist (e->to, tmp);
293 
294  tag = headers ? strtok_r (headers, "&", &p) : NULL;
295 
296  for (; tag; tag = strtok_r (NULL, "&", &p))
297  {
298  if ((value = strchr (tag, '=')))
299  *value++ = '\0';
300  if (!value || !*value)
301  continue;
302 
303  if (url_pct_decode (tag) < 0)
304  goto out;
305  if (url_pct_decode (value) < 0)
306  goto out;
307 
308  /* Determine if this header field is on the allowed list. Since Mutt
309  * interprets some header fields specially (such as
310  * "Attach: ~/.gnupg/secring.gpg"), care must be taken to ensure that
311  * only safe fields are allowed.
312  *
313  * RFC2368, "4. Unsafe headers"
314  * The user agent interpreting a mailto URL SHOULD choose not to create
315  * a message if any of the headers are considered dangerous; it may also
316  * choose to create a message with only a subset of the headers given in
317  * the URL.
318  */
319  if (mutt_matches_list (tag, MailtoAllow))
320  {
321  if (!ascii_strcasecmp (tag, "body"))
322  {
323  if (body)
324  mutt_str_replace (body, value);
325  }
326  else
327  {
328  char *scratch;
329  size_t taglen = mutt_strlen (tag);
330 
331  safe_asprintf (&scratch, "%s: %s", tag, value);
332  scratch[taglen] = 0; /* overwrite the colon as mutt_parse_rfc822_line expects */
333  value = skip_email_wsp(&scratch[taglen + 1]);
334  mutt_parse_rfc822_line (e, NULL, scratch, value, 1, 0, 1, &last);
335  FREE (&scratch);
336  }
337  }
338  }
339 
340  /* RFC2047 decode after the RFC822 parsing */
341  rfc2047_decode_adrlist (e->from);
342  rfc2047_decode_adrlist (e->to);
343  rfc2047_decode_adrlist (e->cc);
344  rfc2047_decode_adrlist (e->bcc);
345  rfc2047_decode_adrlist (e->reply_to);
346  rfc2047_decode_adrlist (e->mail_followup_to);
347  rfc2047_decode_adrlist (e->return_path);
348  rfc2047_decode_adrlist (e->sender);
349  rfc2047_decode (&e->x_label);
350  rfc2047_decode (&e->subject);
351 
352  rc = 0;
353 
354 out:
355  FREE (&tmp);
356  return rc;
357 }
