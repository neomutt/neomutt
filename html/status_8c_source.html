49 /* %b = number of incoming folders with unread messages [option]
50  * %d = number of deleted messages [option]
51  * %f = full mailbox path
52  * %F = number of flagged messages [option]
53  * %h = hostname
54  * %l = length of mailbox (in bytes) [option]
55  * %m = total number of messages [option]
56  * %M = number of messages shown (virtual message count when limiting) [option]
57  * %n = number of new messages [option]
58  * %o = number of old unread messages [option]
59  * %p = number of postponed messages [option]
60  * %P = percent of way through index
61  * %r = readonly/wontwrite/changed flag
62  * %s = current sorting method ($sort)
63  * %S = current aux sorting method ($sort_aux)
64  * %t = # of tagged messages [option]
65  * %u = number of unread messages [option]
66  * %v = Mutt version
67  * %V = currently active limit pattern [option] */
68 static const char *
69 status_format_str (char *buf, size_t buflen, size_t col, int cols, char op, const char *src,
70  const char *prefix, const char *ifstring,
71  const char *elsestring,
72  unsigned long data, format_flag flags)
73 {
74  char fmt[SHORT_STRING], tmp[SHORT_STRING], *cp;
75  int count, optional = (flags & MUTT_FORMAT_OPTIONAL);
76  MUTTMENU *menu = (MUTTMENU *) data;
77 
78  *buf = 0;
79  switch (op)
80  {
81  case 'b':
82  if (!optional)
83  {
84  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
85  snprintf (buf, buflen, fmt, mutt_buffy_check (0));
86  }
87  else if (!mutt_buffy_check (0))
88  optional = 0;
89  break;
90 
91  case 'd':
92  if (!optional)
93  {
94  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
95  snprintf (buf, buflen, fmt, Context ? Context->deleted : 0);
96  }
97  else if (!Context || !Context->deleted)
98  optional = 0;
99  break;
100 
101  case 'f':
102  {
103 #ifdef USE_NOTMUCH
104  char *p;
105  if (Context && Context->magic == MUTT_NOTMUCH &&
106  (p = nm_get_description(Context)))
107  strfcpy(tmp, p, sizeof (tmp));
108  else
109 #endif
110 #ifdef USE_COMPRESSED
111  if (Context && Context->compress_info && Context->realpath) {
112  strfcpy (tmp, Context->realpath, sizeof (tmp));
113  mutt_pretty_mailbox (tmp, sizeof (tmp));
114  } else
115 #endif
116  if (Context && Context->path)
117  {
118  strfcpy (tmp, Context->path, sizeof (tmp));
119  mutt_pretty_mailbox (tmp, sizeof (tmp));
120  }
121  else
122  strfcpy (tmp, _("(no mailbox)"), sizeof (tmp));
123 
124  snprintf (fmt, sizeof(fmt), "%%%ss", prefix);
125  snprintf (buf, buflen, fmt, tmp);
126  break;
127  }
128  case 'F':
129  if (!optional)
130  {
131  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
132  snprintf (buf, buflen, fmt, Context ? Context->flagged : 0);
133  }
134  else if (!Context || !Context->flagged)
135  optional = 0;
136  break;
137 
138  case 'h':
139  snprintf (fmt, sizeof (fmt), "%%%ss", prefix);
140  snprintf (buf, buflen, fmt, NONULL(Hostname));
141  break;
142 
143  case 'l':
144  if (!optional)
145  {
146  snprintf (fmt, sizeof (fmt), "%%%ss", prefix);
147  mutt_pretty_size (tmp, sizeof (tmp), Context ? Context->size : 0);
148  snprintf (buf, buflen, fmt, tmp);
149  }
150  else if (!Context || !Context->size)
151  optional = 0;
152  break;
153 
154  case 'L':
155  if (!optional)
156  {
157  snprintf (fmt, sizeof (fmt), "%%%ss", prefix);
158  mutt_pretty_size (tmp, sizeof (tmp), Context ? Context->vsize: 0);
159  snprintf (buf, buflen, fmt, tmp);
160  }
161  else if (!Context || !Context->pattern)
162  optional = 0;
163  break;
164 
165  case 'm':
166  if (!optional)
167  {
168  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
169  snprintf (buf, buflen, fmt, Context ? Context->msgcount : 0);
170  }
171  else if (!Context || !Context->msgcount)
172  optional = 0;
173  break;
174 
175  case 'M':
176  if (!optional)
177  {
178  snprintf (fmt, sizeof(fmt), "%%%sd", prefix);
179  snprintf (buf, buflen, fmt, Context ? Context->vcount : 0);
180  }
181  else if (!Context || !Context->pattern)
182  optional = 0;
183  break;
184 
185  case 'n':
186  if (!optional)
187  {
188  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
189  snprintf (buf, buflen, fmt, Context ? Context->new : 0);
190  }
191  else if (!Context || !Context->new)
192  optional = 0;
193  break;
194 
195  case 'o':
196  if (!optional)
197  {
198  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
199  snprintf (buf, buflen, fmt, Context ? Context->unread - Context->new : 0);
200  }
201  else if (!Context || !(Context->unread - Context->new))
202  optional = 0;
203  break;
204 
205  case 'p':
206  count = mutt_num_postponed (0);
207  if (!optional)
208  {
209  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
210  snprintf (buf, buflen, fmt, count);
211  }
212  else if (!count)
213  optional = 0;
214  break;
215 
216  case 'P':
217  if (!menu)
218  break;
219  if (menu->top + menu->pagelen >= menu->max)
220  cp = menu->top ? "end" : "all";
221  else
222  {
223  count = (100 * (menu->top + menu->pagelen)) / menu->max;
224  snprintf (tmp, sizeof (tmp), "%d%%", count);
225  cp = tmp;
226  }
227  snprintf (fmt, sizeof (fmt), "%%%ss", prefix);
228  snprintf (buf, buflen, fmt, cp);
229  break;
230 
231  case 'r':
232  {
233  size_t i = 0;
234 
235  if (Context)
236  {
237  i = option(OPTATTACHMSG) ? 3 : ((Context->readonly ||
238  Context->dontwrite) ? 2 : (Context->changed ||
239  /* deleted doesn't necessarily mean changed in IMAP */
240  (Context->magic != MUTT_IMAP &&
241  Context->deleted)) ? 1 : 0);
242  }
243 
244  if (!StChars || !StChars->len)
245  buf[0] = 0;
246  else if (i >= StChars->len)
247  snprintf (buf, buflen, "%s", StChars->chars[0]);
248  else
249  snprintf (buf, buflen, "%s", StChars->chars[i]);
250  break;
251  }
252 
253  case 's':
254  snprintf (fmt, sizeof (fmt), "%%%ss", prefix);
255  snprintf (buf, buflen, fmt,
256  get_sort_str (tmp, sizeof (tmp), Sort));
257  break;
258 
259  case 'S':
260  snprintf (fmt, sizeof (fmt), "%%%ss", prefix);
261  snprintf (buf, buflen, fmt,
262  get_sort_str (tmp, sizeof (tmp), SortAux));
263  break;
264 
265  case 't':
266  if (!optional)
267  {
268  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
269  snprintf (buf, buflen, fmt, Context ? Context->tagged : 0);
270  }
271  else if (!Context || !Context->tagged)
272  optional = 0;
273  break;
274 
275  case 'u':
276  if (!optional)
277  {
278  snprintf (fmt, sizeof (fmt), "%%%sd", prefix);
279  snprintf (buf, buflen, fmt, Context ? Context->unread : 0);
280  }
281  else if (!Context || !Context->unread)
282  optional = 0;
283  break;
284 
285  case 'v':
286  snprintf (fmt, sizeof (fmt), "NeoMutt %%s%%s");
287  snprintf (buf, buflen, fmt, PACKAGE_VERSION, GitVer);
288  break;
289 
290  case 'V':
291  if (!optional)
292  {
293  snprintf (fmt, sizeof(fmt), "%%%ss", prefix);
294  snprintf (buf, buflen, fmt, (Context && Context->pattern) ? Context->pattern : "");
295  }
296  else if (!Context || !Context->pattern)
297  optional = 0;
298  break;
299 
300  case 0:
301  *buf = 0;
302  return (src);
303 
304  default:
305  snprintf (buf, buflen, "%%%s%c", prefix, op);
306  break;
307  }
308 
309  if (optional)
310  _menu_status_line (buf, buflen, col, cols, menu, ifstring);
311  else if (flags & MUTT_FORMAT_OPTIONAL)
312  _menu_status_line (buf, buflen, col, cols, menu, elsestring);
313 
314  return (src);
315 }
