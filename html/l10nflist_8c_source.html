147 /* Return number of bits set in X. */
148 static int pop PARAMS ((int x));
149 
150 static inline int
151 pop (x)
152  int x;
153 {
154  /* We assume that no more than 16 bits are used. */
155  x = ((x & ~0x5555) >> 1) + (x & 0x5555);
156  x = ((x & ~0x3333) >> 2) + (x & 0x3333);
157  x = ((x >> 4) + x) & 0x0f0f;
158  x = ((x >> 8) + x) & 0xff;
159 
160  return x;
161 }
