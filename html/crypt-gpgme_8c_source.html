320 /* Return true whe validity of KEY is sufficient. */
321 static int crypt_id_is_strong (crypt_key_t *key)
322 {
323  unsigned int is_strong = 0;
324 
325  if ((key->flags & KEYFLAG_ISX509))
326  return 1;
327 
328  switch (key->validity)
329  {
330  case GPGME_VALIDITY_UNKNOWN:
331  case GPGME_VALIDITY_UNDEFINED:
332  case GPGME_VALIDITY_NEVER:
333  case GPGME_VALIDITY_MARGINAL:
334  is_strong = 0;
335  break;
336 
337  case GPGME_VALIDITY_FULL:
338  case GPGME_VALIDITY_ULTIMATE:
339  is_strong = 1;
340  break;
341  }

352 /* Return a bit vector describing how well the addresses ADDR and
353  U_ADDR match and whether KEY is valid. */
354 static int crypt_id_matches_addr (ADDRESS *addr, ADDRESS *u_addr,
355  crypt_key_t *key)
356 {
357  int rv = 0;
358 
359  if (crypt_id_is_valid (key))
360  rv |= CRYPT_KV_VALID;
361 
362  if (crypt_id_is_strong (key))
363  rv |= CRYPT_KV_STRONGID;
364 
365  if (addr->mailbox && u_addr->mailbox
366  && mutt_strcasecmp (addr->mailbox, u_addr->mailbox) == 0)
367  rv |= CRYPT_KV_ADDR;
368 
369  if (addr->personal && u_addr->personal
370  && mutt_strcasecmp (addr->personal, u_addr->personal) == 0)
371  rv |= CRYPT_KV_STRING;
372 
373  return rv;
374 }

429 /* Create a new GPGME Data object from the mail body A. With CONVERT
430  passed as true, the lines are converted to CR,LF if required.
431  Return NULL on error or the gpgme_data_t object on success. */
432 static gpgme_data_t body_to_data_object (BODY *a, int convert)
433 {
434  char tempfile[_POSIX_PATH_MAX];
435  FILE *fptmp;
436  int err = 0;
437  gpgme_data_t data;
438 
439  mutt_mktemp (tempfile, sizeof (tempfile));
440  fptmp = safe_fopen (tempfile, "w+");
441  if (!fptmp)
442  {
443  mutt_perror (tempfile);
444  return NULL;
445  }
446 
447  mutt_write_mime_header (a, fptmp);
448  fputc ('\n', fptmp);
449  mutt_write_mime_body (a, fptmp);
450 
451  if (convert)
452  {
453  int c, hadcr = 0;
454  unsigned char buf[1];
455 
456  data = create_gpgme_data ();
457  rewind (fptmp);
458  while ((c = fgetc (fptmp)) != EOF)
459  {
460  if (c == '\r')
461  hadcr = 1;
462  else
463  {
464  if (c == '\n' && !hadcr)
465  {
466  buf[0] = '\r';
467  gpgme_data_write (data, buf, 1);
468  }
469 
470  hadcr = 0;
471  }
472  /* FIXME: This is quite suboptimal */
473  buf[0] = c;
474  gpgme_data_write (data, buf, 1);
475  }
476  safe_fclose (&fptmp);
477  gpgme_data_seek (data, 0, SEEK_SET);
478  }
479  else
480  {
481  safe_fclose (&fptmp);
482  err = gpgme_data_new_from_file (&data, tempfile, 1);
483  }
484  unlink (tempfile);
485  if (err)
486  {
487  mutt_error (_("error allocating data object: %s\n"), gpgme_strerror (err));
488  return NULL;
489  }

631 /* Create a GpgmeRecipientSet from the keys in the string KEYLIST.
632  The keys must be space delimited. */
633 static gpgme_key_t *create_recipient_set (const char *keylist,
634  gpgme_protocol_t protocol)
635 {
636  int err;
637  const char *s;
638  char buf[100];
639  int i;
640  gpgme_key_t *rset = NULL;
641  unsigned int rset_n = 0;
642  gpgme_key_t key = NULL;
643  gpgme_ctx_t context = NULL;
644 
645  err = gpgme_new (&context);
646  if (! err)
647  err = gpgme_set_protocol (context, protocol);
648 
649  if (! err)
650  {
651  s = keylist;
652  do {
653  while (*s == ' ')
654  s++;
655  for (i=0; *s && *s != ' ' && i < sizeof(buf)-1;)
656  buf[i++] = *s++;
657  buf[i] = 0;
658  if (*buf)
659  {
660  if (i>1 && buf[i-1] == '!')
661  {
662  /* The user selected to override the validity of that
663  key. */
664  buf[i-1] = 0;
665 
666  err = gpgme_get_key (context, buf, &key, 0);
667  if (! err)
668  key->uids->validity = GPGME_VALIDITY_FULL;
669  buf[i-1] = '!';
670  }
671  else
672  err = gpgme_get_key (context, buf, &key, 0);
673 
674  safe_realloc (&rset, sizeof (*rset) * (rset_n + 1));
675  if (! err)
676  rset[rset_n++] = key;
677  else
678  {
679  mutt_error (_("error adding recipient `%s': %s\n"),
680  buf, gpgme_strerror (err));
681  rset[rset_n] = NULL;
682  free_recipient_set (&rset);
683  gpgme_release (context);
684  return NULL;
685  }
686  }
687  } while (*s);
688  }
689 
690  /* NULL terminate. */
691  safe_realloc (&rset, sizeof (*rset) * (rset_n + 1));
692  rset[rset_n++] = NULL;
693 
694  if (context)
695  gpgme_release (context);
696 
697  return rset;
698 }

701 /* Make sure that the correct signer is set. Returns 0 on success. */
702 static int set_signer (gpgme_ctx_t ctx, int for_smime)
703 {
704  char *signid = for_smime ? SmimeDefaultKey: PgpSignAs;
705  gpgme_error_t err;
706  gpgme_ctx_t listctx;
707  gpgme_key_t key, key2;
708 
709  if (!signid || !*signid)
710  return 0;
711 
712  listctx = create_gpgme_context (for_smime);
713  err = gpgme_op_keylist_start (listctx, signid, 1);
714  if (!err)
715  err = gpgme_op_keylist_next (listctx, &key);
716  if (err)
717  {
718  gpgme_release (listctx);
719  mutt_error (_("secret key `%s' not found: %s\n"),
720  signid, gpgme_strerror (err));
721  return -1;
722  }
723  err = gpgme_op_keylist_next (listctx, &key2);
724  if (!err)
725  {
726  gpgme_key_unref (key);
727  gpgme_key_unref (key2);
728  gpgme_release (listctx);
729  mutt_error (_("ambiguous specification of secret key `%s'\n"),
730  signid);
731  return -1;
732  }
733  gpgme_op_keylist_end (listctx);
734  gpgme_release (listctx);
735 
736  gpgme_signers_clear (ctx);
737  err = gpgme_signers_add (ctx, key);
738  gpgme_key_unref (key);
739  if (err)
740  {
741  mutt_error (_("error setting secret key `%s': %s\n"),
742  signid, gpgme_strerror (err));
743  return -1;
744  }
745  return 0;

1662  IS_SMIME) with body A described further by state S. Write
1663  plaintext out to file FPOUT and return a new body. For PGP returns
1664  a flag in R_IS_SIGNED to indicate whether this is a combined
1665  encrypted and signed message, for S/MIME it returns true when it is
1666  not a encrypted but a signed message. */

1668  int *r_is_signed)
1669 {

1671  BODY *tattach;
1672  int err = 0;
1673  gpgme_ctx_t ctx;
1674  gpgme_data_t ciphertext, plaintext;
1675  int maybe_signed = 0;
1676  int anywarn = 0;
1677  int sig_stat = 0;
1678 
1679  if (r_is_signed)
1680  *r_is_signed = 0;
1681 
1682  ctx = create_gpgme_context (is_smime);
1683 
1684  restart:
1685  /* Make a data object from the body, create context etc. */
1686  ciphertext = file_to_data_object (s->fpin, a->offset, a->length);
1687  if (!ciphertext)
1688  return NULL;
1689  plaintext = create_gpgme_data ();
1690 
1691  /* Do the decryption or the verification in case of the S/MIME hack. */
1692  if ((! is_smime) || maybe_signed)
1693  {
1694  if (! is_smime)
1695  err = gpgme_op_decrypt_verify (ctx, ciphertext, plaintext);
1696  else if (maybe_signed)
1697  err = gpgme_op_verify (ctx, ciphertext, NULL, plaintext);
1698 
1699  if (err == GPG_ERR_NO_ERROR)
1700  {
1701  /* Check whether signatures have been verified. */
1702  gpgme_verify_result_t verify_result = gpgme_op_verify_result (ctx);
1703  if (verify_result->signatures)
1704  sig_stat = 1;
1705  }
1706  }
1707  else
1708  err = gpgme_op_decrypt (ctx, ciphertext, plaintext);
1709  gpgme_data_release (ciphertext);
1710  if (err)
1711  {
1712  if (is_smime && !maybe_signed
1713  && gpg_err_code (err) == GPG_ERR_NO_DATA)
1714  {

1719  it has not been set. */

1721

1723  if (!result->unsupported_algorithm)
1724  {

1732  {
1733  char buf[200];
1734

1738  state_attach_puts (buf, s);
1739  }
1740  gpgme_data_release (plaintext);
1741  gpgme_release (ctx);
1742  return NULL;
1743  }
1744  mutt_need_hard_redraw ();

1747  otherwise read_mime_header has a hard time parsing the message. */
1748  if (data_object_to_stream (plaintext, fpout))
1749  {
1750  gpgme_data_release (plaintext);
1751  gpgme_release (ctx);
1752  return NULL;
1753  }
1754  gpgme_data_release (plaintext);
1755 
1756  a->is_signed_data = 0;
1757  if (sig_stat)
1758  {
1759  int res, idx;
1760  int anybad = 0;
1761 
1762  if (maybe_signed)
1763  a->is_signed_data = 1;
1764  if(r_is_signed)
1765  *r_is_signed = -1; /* A signature exists. */
1766 
1767  if ((s->flags & MUTT_DISPLAY))
1768  state_attach_puts (_("[-- Begin signature "
1769  "information --]\n"), s);
1770  for(idx = 0; (res = show_one_sig_status (ctx, idx, s)) != -1; idx++)
1771  {
1772  if (res == 1)
1773  anybad = 1;
1774  else if (res == 2)
1775  anywarn = 1;
1776  }
1777  if (!anybad && idx && r_is_signed && *r_is_signed)
1778  *r_is_signed = anywarn? 2:1; /* Good signature. */
1779 
1780  if ((s->flags & MUTT_DISPLAY))
1781  state_attach_puts (_("[-- End signature "
1782  "information --]\n\n"), s);
1783  }
1784  gpgme_release (ctx); ctx = NULL;
1785 
1786  fflush (fpout);
1787  rewind (fpout);
1788  tattach = mutt_read_mime_header (fpout, 0);
1789  if (tattach)
1790  {
1791  /*
1792  * Need to set the length of this body part.
1793  */
1794  fstat (fileno (fpout), &info);
1795  tattach->length = info.st_size - tattach->offset;
1796 
1797  tattach->warnsig = anywarn;
1798 
1799  /* See if we need to recurse on this MIME part. */
1800  mutt_parse_part (fpout, tattach);
1801  }
1802 
1803  return tattach;
